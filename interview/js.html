# 1. bind 属性丢失问题

function a() {}
a.abc = 123

const b = a.bind()

console.log(b === a) // false
console.log(a.abc) // 123
console.log(b.abc) // undefined

拓展：

在vue中实现：input 失去焦点后，调用 debounce 函数返回的 cancel 方法取消搜索， 然而，vue 使用 bind 方法将 vue 实例绑定到函数，从而丢失了 cancel 方法。

this.querySearch = methods.querySearch.bind(vue实例) 

<script>
methods: {
  querySearch: debounce(async function(query, cb) {
    cb(await search(query))
  }, 1000)
  cancel() {
    console.log(this.querySearch.cancel) // undefined
  }
}
</script>

可将 querySearch 方法放入 data 中。

<script>
data() {
  return {
    querySearch: debounce(async function(query, cb) {
      cb(await search(query))
    }, 1000)
  }
},
methods: {
  cancel() {
    console.log(this.querySearch.cancel) // undefined
  }
}
</script>

# 2. 如何将 class 转换为 function

class Example {
  constructor(name) {
    this.name = name
  }
  func() {
    console.log(this.name)
  }
}

// 不可枚举
var e = new Example('abc')
for (const key in e) {
  console.log(key)
}

// 报错
new e.func() 

'use strict';

function Example(name) {
  if (!(this instanceof Example)) {
    throw new TypeError(`Class constructor Example cannot be invoked without 'new'`)
  }
  this.name = name
}

Object.defineProperty(Example.prototype, 'func', {
  value: function() {
    if (!(this instanceof Example)) {
      throw new TypeError(`.func is not a constructor`)
    }
    console.log(this.name)
  },
  enumerable: false, // 不可枚举
})
