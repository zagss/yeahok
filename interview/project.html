# 💼 项目实战

# 1. 高亮关键字

模糊搜索高亮关键字

function setHearoHTML(heroes) {
  // 正则匹配搜索的字符
  const key = 'sk'
  let reg = new RegExp(key, 'gi')

  doms.ul.innerHTML = heros.map(h => {
    var name = h.cname
    name = name.replace(reg, key => `<em>${key}</em>`)
    return `<li>
      <img />
      <span>${name}</span>
    </li>`
  })
}

# 2. 跨标签页通信

常见方案：

  • BroadCast Channel
  • Service Worker
  • LocalStorage window.onstorage
  • Shared Worker 定时器轮询（setInterval）
  • IndexedDB 定时器轮询（setInterval）
  • cookie 定时器轮询（setInterval）
  • window.open window.postMessage
  • Websocket

消息发送和监听：

// crossTagMsg.js
const channel = new BroadcastChannel('demo')

export function sendMsg(type, content) {
  channel.postMessage({
    type,
    content
  })
}

export function listenMsg(callback) {
  const handler = e => {
    callback && callback(e.data)
  }
  channel.addEventListener('message', handler)
  return () => {
    channel.removeEventListener('message', handler)
  }
}

使用：

<script>
function handleSubmit() {
  sendMsg('add-item', { ...item.value })
}

const cancelListen = listenMsg(msg => {
  if (msg.type === 'add-item') {
    ...
  }
})

onUnmounted(cancelListen)
</script>

# 3.自定义ref

使用自定义ref实现防抖

<template>
  <div>
    <input v-model="text" />
    <p>{{ text }}</p>
  </div>
</template>

<script setup>
import { debounceRef } from './debounceRef'
const text = debounceRef('', 500)
</script>

debounceRef.js:

import { customRef } from 'vue'

export function debounceRef(value, duration = 1000) {
  let timer;
  return customRef((track, trigger) => {
    return {
      get() {
        // 收集依赖
        track()
        return value
      },
      set(val) {
        clearTimeout(timer)
        timer = setTimeout(() => {
          // 派发更新
          trigger()
          value = val
        }, duration)
      }
    }
  })
}
