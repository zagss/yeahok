# DOM获取元素尺寸的属性有哪些？

  • offsetWidth: 元素的可见宽度，包括 border、padding、width
  • offsetHeight：元素的可见高度，包括 border、padding、height
  • clientWidth：元素的可见宽度，包括padding、不包括border、margin
  • clientHeight：元素的可见高度，包括padding、不包括border、margin
  • scrollWidth：元素的滚动宽度，包括padding、不包括border、margin
  • scrollHeight：元素的滚动高度，包括padding、不包括border、margin
  • offsetTop：元素距离父元素顶部的距离。
  • offsetLeft：元素距离父元素左侧的距离。

# 使用defer 优化白屏时间

// useDefer.js
export function useDefer() {
  const frameCount = ref(0)
  function updateFrameCount() {
    requestAnimationFrame(() => {
      frameCount.value++
      updateFrameCount()
    })
  }
  updateFrameCount()
  return function defer(n) {
    return frameCount.value >= n
  }
}

<template>
  <div v-for="n in 100">
    <heavy-comp v-if="defer(n)"></heavy-comp>
  </div>
</template>

# 将传入的函数放入微队列执行

function runMicrotask(fn) {
  if (typeof Promise === 'function') {
    Promise.resolve().then(func)
    return
  }
  if (typeof MutationObserver === 'function') {
    var ob = new MutationObserver(fn)
    var node = document.createTextNode('')
    ob.observe(node, { characterData: true })
    node.data = 1
    return
  }
  if (process && typeof process.nextTick === 'function') {
    process.nextTick(fn)
    return
  }
}

# 两个超过整数存储范围的大整数求和

function sum(a: string, b: string) {
  const len = Math.max(a.length, b.length)
  a = a.padStart(len, '0')
  b = b.padStart(len, '0')
  let carry = 0
  let result = ''
  for (let i = len - 1; i >= 0; i--) {
    const sum = +a[i] + +b[i] + carry
    // 保留个位(sum % 10)计算结果，存储进位carry
    result = (sum % 10) + result
    // 向下取整，0 / 1
    carry = Math.floor(sum / 10)
  }
  if (carry) {
    result = carry + result
  }
  return result
}

# 并发请求

/**
 * 并发请求
 * @param {string[]} urls 待请求的url数组
 * @param {number} maxNum 最大并发数
 * @returns 
 */
function concurRequest(urls, maxNum) {
  return new Promise(resolve => {
    if (!urls.length) {
      resolve([])
      return
    }
    const result = []
    let index = 0 // 下一个请求的下标
    let count = 0 // 当前完成请求数量
    async function request() {
      if (index === urls.length) {
        return
      }
      const i = index
      const url = urls[index]
      index++
      try {
        const resp = await fetch(url)
        result[i] = resp
      } catch (error) {
        result[i] = error
      } finally {
        count++
        if (count === urls.length) {
          resolve(result)
        }
        request()
      }
    }
    const times = Math.min(maxNum, urls.length)
    for (let i = 0; i < times; i++) {
      request()
    }
  })
}

# 给fetch 添加超时功能

function createFetchWithTimeout(timeout = 10000) {
  return function (url, options) {
    return new Promise((resolve, reject) => {
      // 创建信号控制器
      const singalController = new AbortController()
      fetch(url, {
        ...options,
        signal: singalController.signal,
      }).then(resolve, reject)
      setTimeout(() => {
        reject(new Error('fetch timeout'))
        // 取消请求
        singalController.abort()
      }, timeout)
    })
  }
}

const request = createFetchWithTimeout(10000)
