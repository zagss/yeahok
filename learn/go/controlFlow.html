# 五、流程控制

# 1. if else 分支结构

在 if 表达式前添加一个执行语句

func ifDemo() {
  if score := 65; score >= 90 {
    fmt.Println("A")
  } else if score > 75 {
    fmt.Println("B")
  } else {
    fmt.Println("C")
  }
}

# 2. for 循环结构

for 循环的初始语句可以被忽略，但是初始语句后的分号必须要写:

func forDemo() {
  i := 0
  for ; i < 10; i++ {
    fmt.Println(i)
  }
}

for循环的初始语句和结束语句都可以省略，类似 while :

func forDemo() {
  i := 0
  for i < 10 {
    fmt.Println(i)
    i++
  }
}

无限循环：

for {
  ...
}

# 3. for range 键值循环

for range 可用来遍历数组、切片、字符串、map、channel 等：

  • 数组、切片、字符串返回 索引和值
  • map 返回 键和值
  • channel 只返回 通道内的值

# 4. switch case

一个分支可以有多个值：

func switchDemo() {
  switch n := 7; n {
    case 1, 3, 5, 7, 9:
      fmt.Println("奇数")
    case 2, 4, 6, 8:
      fmt.Println("偶数")
    default:
      fmt.Println(n)
  }
}

fallthrough 语法可以执行满足条件的 case 的下一个 case

func switchDemo() {
	s := "a"
	switch {
	case s == "a":
		fmt.Println("a")
		fallthrough
	case s == "b":
		fmt.Println("b")
	case s == "c":
		fmt.Println("c")
	default:
		fmt.Println("...")
	}
}
// a
// b

# 5. goto 跳转到指定标签

goto 语句通过标签进行代码间的无条件跳转，能简化代码：

// 示例：
func gotoDemo1() {
  var breakFlag bool
  for i := 0; i < 10; i++ {
    for j := 0; j < 10; j++ {
      if j == 2 {
        // 设置退出标签
        breakFlag = true
        break
      }
    }
    // 外层for循环判断
    if breakFlag {
      break
    }
  }
}

// goto 简化：
func gotoDemo2() {
  for i := 0; i < 10; i++ {
    for j := 0; j < 10; j++ {
      if j == 2 {
        goto breakTag
      }
      fmt.Printf("%v-%v\n", i, j)
    }
  }
  return
breakTag:
  fmt.Println("结束for循环")
}

// 0-0
// 0-1
// 结束for循环

# 6. break 跳出循环

break 语句可以结束 for 、switch 和 select 的代码块。

func breakDemo() {
breakFlag:
  for i := 0; i < 10; i++ {
    for j :=0; j < 10; j++ {
      if j == 2 {
        break breakFlag
      }
      fmt.Printf("%v-%v\n", i, j)
    }
  }
  fmt.Println("...")
}

// 0-0
// 0-1
// ...

# 7. continue 继续下次循环

continue 语句可以结束当前循环，开始下一次的循环迭代过程，仅限在 for 循环内使用。

func continueDemo() {
forloop1:
	for i := 0; i < 3; i++ {
		for j := 0; j < 3; j++ {
			if i == 1 && j == 1 {
				continue forloop1
			}
			fmt.Printf("%v-%v\n", i, j)
		}
	}
}

// 0-0
// 0-1
// 0-2
// 1-0
// 2-0
// 2-1
// 2-2

# 8. 九九乘法表

func demo() {
  // 第几行
  for y := 1; y <= 9; y++ {
    // 第几列
    for x := 1; x <= y; x++ {
      fmt.Printf("%d*%d=%d ", x, y, x * y)
    }
    // 回车
    fmt.Println()
  }
}
